## TransitOps API Test Examples

# ------------------------------
# Auth API Examples
# ------------------------------

# 1. Signup - Create a fleet manager
curl -X POST http://localhost:5000/api/auth/signup \
  -H "Content-Type: application/json" \
  -d '{
    "name": "John Doe",
    "email": "john@example.com",
    "password": "password123",
    "role": "fleet_manager"
  }'

# 2. Signup - Create a driver (will fail with EMAIL_EXISTS if using same email)
curl -X POST http://localhost:5000/api/auth/signup \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Jane Smith",
    "email": "jane@example.com",
    "password": "password123",
    "role": "driver"
  }'

# 3. Login - Get JWT token (replace with actual email/password)
curl -X POST http://localhost:5000/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "email": "john@example.com",
    "password": "password123"
  }'

# 4. Get Current User (me) - Replace <YOUR_TOKEN> with token from login
curl -X GET http://localhost:5000/api/auth/me \
  -H "Authorization: Bearer <YOUR_TOKEN>"

# 5. Logout - Replace <YOUR_TOKEN>
curl -X POST http://localhost:5000/api/auth/logout \
  -H "Authorization: Bearer <YOUR_TOKEN>"

# ------------------------------
# Trip Management API Examples (Replace <YOUR_TOKEN>, <TRIP_ID>, <VEHICLE_ID>, <DRIVER_ID> as needed)
# ------------------------------

# 6. Create Trip (fleet_manager or driver)
curl -X POST http://localhost:5000/api/trips \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer <YOUR_TOKEN>" \
  -d '{
    "source": "Warehouse A",
    "destination": "Store B",
    "vehicle_id": <VEHICLE_ID>,
    "driver_id": <DRIVER_ID>,
    "cargo_weight": 1500,
    "planned_distance": 150
  }'

# 7. Get All Trips (all roles) - Optional query params: status, vehicle_id, driver_id
curl -X GET http://localhost:5000/api/trips \
  -H "Authorization: Bearer <YOUR_TOKEN>"

# 8. Get Trip by ID (all roles)
curl -X GET http://localhost:5000/api/trips/<TRIP_ID> \
  -H "Authorization: Bearer <YOUR_TOKEN>"

# 9. Dispatch Trip (fleet_manager or driver)
curl -X PATCH http://localhost:5000/api/trips/<TRIP_ID>/dispatch \
  -H "Authorization: Bearer <YOUR_TOKEN>"

# 10. Complete Trip (fleet_manager or driver)
curl -X PATCH http://localhost:5000/api/trips/<TRIP_ID>/complete \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer <YOUR_TOKEN>" \
  -d '{
    "actual_distance": 155,
    "fuel_consumed": 35.5
  }'

# 11. Cancel Trip (fleet_manager or driver)
curl -X PATCH http://localhost:5000/api/trips/<TRIP_ID>/cancel \
  -H "Authorization: Bearer <YOUR_TOKEN>"

# 12. Delete Trip (fleet_manager only, only if status=draft)
curl -X DELETE http://localhost:5000/api/trips/<TRIP_ID> \
  -H "Authorization: Bearer <YOUR_TOKEN>"

# ------------------------------
# Maintenance API Examples (Replace <YOUR_TOKEN>, <MAINTENANCE_ID>, <VEHICLE_ID> as needed)
# ------------------------------

# 13. Create Maintenance Log (fleet_manager only)
curl -X POST http://localhost:5000/api/maintenance \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer <YOUR_TOKEN>" \
  -d '{
    "vehicle_id": <VEHICLE_ID>,
    "description": "Oil change and tire rotation",
    "cost": 150.00,
    "priority": "medium"
  }'

# 14. Get All Maintenance Logs (fleet_manager or financial_analyst) - Optional query params: status, vehicle_id
curl -X GET http://localhost:5000/api/maintenance \
  -H "Authorization: Bearer <YOUR_TOKEN>"

# 15. Get Maintenance Log by ID (fleet_manager only)
curl -X GET http://localhost:5000/api/maintenance/<MAINTENANCE_ID> \
  -H "Authorization: Bearer <YOUR_TOKEN>"

# 16. Close Maintenance Log (fleet_manager only)
curl -X PATCH http://localhost:5000/api/maintenance/<MAINTENANCE_ID>/close \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer <YOUR_TOKEN>" \
  -d '{
    "closing_notes": "Maintenance completed successfully"
  }'
